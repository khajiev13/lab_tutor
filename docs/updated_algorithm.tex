\begin{algorithm}
\caption{\textbf{Phase 0: Agentic Book Discovery, Scoring \& Acquisition}}
\begin{algorithmic}[1]
\bfseries
\Require Course context $\mathcal{D}$ (title, description, syllabus with keywords), Course level $\ell$
\Ensure Downloaded books $\mathcal{B}_{all}$ with PDFs

\State \textbf{// Stage 1: Query Generation \& Parallel Discovery}
\State $\{q_1, \dots, q_{12}\} \leftarrow \text{LLM}_{gen}(\mathcal{D},\ \text{"Generate search queries"})$

\For{each $q_i$ \textbf{in parallel}}
    \State $R_i \leftarrow \text{GoogleBooksAPI}(q_i) \cup \text{TavilySearch}(q_i)$
    \State $\mathcal{R} \leftarrow \mathcal{R} \cup \text{LLM}_{extract}(R_i)$
\EndFor
\State $\mathcal{B}_{cand} \leftarrow \text{FuzzyDedup}(\mathcal{R})$

\State
\State \textbf{// Stage 2: Research-Augmented Merit Scoring}
\For{each $B_i$ in $\mathcal{B}_{cand}$ \textbf{in parallel}}
    \State $\text{evidence}_i \leftarrow \text{ReActResearch}(B_i,\ \text{tools},\ R_{max}{=}5)$
    \State $(C_{topic}, C_{struc}, C_{scope}, C_{pub}, C_{auth}, C_{time}, C_{prac})$
    \State $\quad \leftarrow \text{LLM}_{score}(B_i,\ \mathcal{D},\ \text{evidence}_i,\ \ell)$
    \State $S_{final}(B_i) \leftarrow \sum_{k} w_k \cdot C_k$
    \Comment{$\mathbf{w} = (0.30, 0.20, 0.15, 0.15, 0.10, 0.10)$}
\EndFor

\State
\State \textbf{// Stage 3: Human-in-the-Loop Review \& Download}
\State $\mathcal{B}_{selected} \leftarrow \text{TeacherReview}\bigl(\text{Rank}(\mathcal{B}_{cand}, S_{final})\bigr)$
\Comment{Teacher selects $\leq 5$ books}
\For{each $B_i$ in $\mathcal{B}_{selected}$ \textbf{in parallel}}
    \State $B_i.\text{pdf} \leftarrow \text{ReActDownload}(B_i)$
\EndFor

\State \Return $\mathcal{B}_{all} \leftarrow \{B_i \in \mathcal{B}_{selected} : B_i.\text{pdf} \neq \emptyset\}$
\end{algorithmic}
\end{algorithm}


\begin{algorithm}
\caption{\textbf{Phase 1: Chunking \& Agentic Extraction for Book Selection}}
\begin{algorithmic}[1]
\bfseries
\Require Course Concepts $\mathcal{Q}$ (from teacher materials), Books $\mathcal{B}_{all}$ (from Phase 0), Course Subject $\sigma$
\Ensure Selected Book $B^*$, Grounded Concepts $O_{final}$

\State \textbf{// Stage 1: Paragraph-Level Chunking \& Scoring}
\For{each $B_i$ in $\mathcal{B}_{all}$}
    \State $S_{toc} \leftarrow \text{ExtractTOC}(B_i)$
    \Comment{4-tier fallback: Bookmarks $\to$ Printed TOC $\to$ Page Scan $\to$ Whole-book}
    \State $\{Ch_1, \dots, Ch_M\} \leftarrow \text{SplitByChapters}(B_i, S_{toc})$
    \State $\mathcal{P}_i \leftarrow \emptyset$
    \For{each Chapter $Ch_j$ in $\{Ch_1, \dots, Ch_M\}$}
        \State $Ch_j \leftarrow \text{CleanText}(Ch_j)$
        \Comment{Remove TOC artifacts, page numbers, noisy headers}
        \State $\mathcal{P}_i \leftarrow \mathcal{P}_i \cup \text{RecursiveSplit}(Ch_j,\ s{=}600,\ o{=}100)$
    \EndFor
    \State $\mathbf{P}_i \leftarrow \text{L2Norm}\bigl(\text{Embed}(\mathcal{P}_i)\bigr)$
\EndFor

\State $\mathbf{Q} \leftarrow \text{L2Norm}\bigl(\text{Embed}(\mathcal{Q})\bigr)$
\Comment{Course concept name embeddings}
\State $\mathcal{S}_{final} \leftarrow \emptyset$

\For{each $B_k$ in $\mathcal{B}_{all}$}
    \State $\mathbf{S} \leftarrow \mathbf{Q} \cdot \mathbf{P}_k^\top$
    \For{each course concept $q_x$}
        \State $\text{sim\_max}(q_x) \leftarrow \max_{p_y} \mathbf{S}[x, y]$
    \EndFor
    \State $\mathcal{S}_{final}[B_k] \leftarrow \frac{1}{|\mathcal{Q}|} \sum_{q_x} \text{sim\_max}(q_x)$
\EndFor

\State $B^* \leftarrow \arg\max_{B_k} (\mathcal{S}_{final})$

\State
\State \textbf{// Stage 2: Agentic Chapter-Level Concept Extraction}
\State $\{Ch_1, \dots, Ch_N\} \leftarrow \text{Chapters}(B^*)$
\Comment{Reuse chapters extracted in Stage 1}

\For{each $Ch_j$ in $\{Ch_1, \dots, Ch_N\}$ \textbf{in parallel}}
    \State \textbf{// Reflection Pattern for Chapter-Level Granularity}
    \State $C_{chapter} \leftarrow \text{LLM}_{gen}(Ch_j,\ \sigma,\ \text{"Extract Concepts"})$
    \State $Feedback \leftarrow \text{LLM}_{eval}(C_{chapter},\ Ch_j,\ \sigma)$

    \While{$Feedback \neq \text{"APPROVED"}\ \land\ \text{iteration} < K_{max}$}
        \State $C_{chapter} \leftarrow \text{LLM}_{gen}(Ch_j,\ C_{chapter},\ Feedback.\text{issues})$
        \State $Feedback \leftarrow \text{LLM}_{eval}(C_{chapter},\ Ch_j,\ \sigma)$
    \EndWhile

    \State $\mathcal{G}_j \leftarrow \text{GroupBySection}(C_{chapter},\ Ch_j.\text{sections})$
    \State $\mathcal{K}_j \leftarrow \text{LLM}_{skills}(\sigma,\ Ch_j,\ \mathcal{G}_j)$
    \Comment{Chapter-level skills extraction}
    \State $\text{Embed}(C_{chapter}.\text{names},\ C_{chapter}.\text{evidence})$
    \Comment{Dual embedding per concept}
\EndFor

\State $O_{final} \leftarrow \bigcup_j \{(\mathcal{G}_j,\ \mathcal{K}_j)\}$

\State \Return $B^*, O_{final}$
\end{algorithmic}
\end{algorithm}
